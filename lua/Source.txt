local dragToggle = false
local dragStart = nil
local startPos = nil
local dragSpeed = 0.1
local ChangedConnection = nil

local function updateInput(input)
	WindowTable.ElBlurUI.Update()
	local delta = input.Position - dragStart
	local position = UDim2.new(
		startPos.X.Scale, 
		startPos.X.Offset + delta.X,
		startPos.Y.Scale, 
		startPos.Y.Offset + delta.Y
	)
	
	game:GetService('TweenService'):Create(MainFrame, TweenInfo.new(dragSpeed), {Position = position}):Play()
	
	if not WindowTable.WindowToggle then
		WindowTable.LastPosition = position 
	end
end

InputFrame.InputBegan:Connect(function(input)
	if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
		if input.UserInputState == Enum.UserInputState.Begin then
			dragToggle = true
			dragStart = input.Position
			startPos = MainFrame.Position
			
			ChangedConnection = input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragToggle = false
					
					if ChangedConnection and ChangedConnection.Connected then
						ChangedConnection:Disconnect()
						ChangedConnection = nil
					end
				end
			end)
		end
	end
end)

local UserInputService = game:GetService("UserInputService")
UserInputService.InputChanged:Connect(function(input)
	if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		if dragToggle and input.UserInputState == Enum.UserInputState.Change then
			updateInput(input)
		end
	end
end)

    WindowTable.Keybind = newKey
end
